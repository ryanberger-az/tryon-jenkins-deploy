#cloud-config
package_upgrade: true
write_files:
  - path: /etc/systemd/system/docker.service.d/docker.conf
    content: |
      [Service]
        ExecStart=
        ExecStart=/usr/bin/dockerd
  - path: /etc/docker/daemon.json
    content: |
      {
        "hosts": ["fd://","tcp://127.0.0.1:2375"]
      }
  - path: /etc/jenkins/plugins.txt
    content: |
      blueocean:1.22.0
      azure-vm-agents:1.4.0
      azure-container-agents:1.1.1
      windows-azure-storage:1.1.5
      azure-credentials:2.0.1
      azure-acs:1.0.1
      cloud-stats:0.25
      configuration-as-code:1.36
      powershell:1.4
runcmd:
  - apt-get update -y
  - apt-get upgrade -y
  - apt install openjdk-8-jre-headless -y
  - apt install maven -y
  - apt install git -y
  - wget -q -O - https://pkg.jenkins.io/debian/jenkins-ci.org.key | sudo apt-key add -
  - sh -c 'echo deb https://pkg.jenkins.io/debian-stable binary/ > /etc/apt/sources.list.d/jenkins.list'
  - apt-get update && apt-get install jenkins -y
  - curl -sSL https://get.docker.com/ | sh
  - usermod -aG docker azureuser
  - usermod -aG docker jenkins
  - service jenkins restart
  - cd /etc/jenkins/
  - git clone https://github.com/jenkinsci/plugin-installation-manager-tool
  - cd plugin-installation-manager-tool
  - wget http://repo.jenkins-ci.org/releases/io/jenkins/plugin-management/plugin-management-cli/1.0.2/plugin-management-cli-1.0.2.jar
  - service jenkins stop
  - java -jar /etc/jenkins/plugin-installation-manager-tool/plugin-management-cli-1.0.2.jar --war /usr/share/jenkins/jenkins.war --plugin-file /etc/jenkins/plugins.txt --plugin-download-directory /etc/jenkins/plugins/
  - cp /etc/jenkins/plugins/* /var/lib/jenkins/plugins/
  - service jenkins start
